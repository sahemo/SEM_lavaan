corr.lv.model <- '

# latent variable definitions 

lat_COM =~ COM1 + COM2 + COM3 
lat_INT =~ INT1 + INT2 + INT3
lat_FAM =~ FAM1 + FAM2 + FAM3
lat_WMC =~ OSPAN + SSPAN


# covariances

VOL ~~ INV
lat_INT ~~ lat_WMC
lat_INT ~~ lat_FAM
lat_INT ~~ lat_COM
lat_COM ~~ lat_FAM
lat_COM ~~ lat_WMC
lat_FAM ~~ lat_WMC



'

myModel <- readLines("model.lav")


cat(myModel, fill = TRUE)
##  # latent variable definitions   lat_COM =~ COM1 + COM2 + COM3

## INV ~ g * lat_WMC + h * lat_INT + i * lat_FAM     # variances and covariances   
## VOL ~~ INV lat_INT ~~ lat_WMC lat_FAM ~~ lat_WMC lat_INT ~~ lat_FAM
